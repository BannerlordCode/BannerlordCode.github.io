{# Tree-structured sidebar for the Bannerlord Modding Wiki. Reads data/section-tree.json (fast JSON) instead of get_section() to keep the Zola build times acceptable with 35k+ pages. #} {%- macro render_nav(current_url, lang) -%} {%- set nav = load_data(path="data/navigation.json") -%} {%- set tree = load_data(path="data/section-tree.json") -%} {%- set_global mapped_url = "/" -%} {%- set_global matched_len = 1 -%} {%- for key, node in nav.routes -%} {%- if current_url is starting_with(key) and key | length > matched_len -%} {%- set_global mapped_url = key -%} {%- set_global matched_len = key | length -%} {%- endif -%} {%- endfor -%} {%- set path_parts = current_url | split(pat="/") -%} {%- set current_version = path_parts | nth(n=1) | default(value="") -%} {%- if nav.meta.versions is containing(current_version) -%} {%- set current_lang = path_parts | nth(n=2) | default(value=lang) -%} {%- if not (nav.meta.languages is containing(current_lang)) -%} {%- set current_lang = lang -%} {%- endif -%} {%- set root_key = "/" ~ current_version ~ "/" ~ current_lang ~ "/" -%} {%- else -%} {%- set root_key = "" -%} {%- endif -%} {%- if root_key and nav.routes[root_key] is defined -%} {%- for group_key, group in nav.groups -%} {%- set_global group_routes = [] -%} {%- if group_key == "start" and nav.routes[root_key] is defined -%} {%- set_global group_routes = group_routes | concat(with=[root_key]) -%} {%- endif -%} {%- if group_key == "cross-version" -%} {%- if nav.routes["/versions/"] is defined -%} {%- set_global group_routes = group_routes | concat(with=["/versions/"]) -%} {%- endif -%} {%- endif -%} {%- for key, node in nav.routes -%} {%- if node.parent == root_key and node.group == group_key -%} {%- set_global group_routes = group_routes | concat(with=[key]) -%} {%- endif -%} {%- endfor -%} {%- if group_routes | length > 0 -%} {%- set_global group_active = false -%} {%- for r in group_routes -%} {%- if mapped_url == r -%} {%- set_global group_active = true -%} {%- endif -%} {%- endfor -%} {%- set is_open = group_active or not group.collapsed -%}
{{ group.label[lang] }}
{%- for route_key in group_routes -%} {%- set node = nav.routes[route_key] -%} {%- set_global sub_routes = [] -%} {%- for child_key in node.children -%} {%- set child = nav.routes[child_key] -%} {%- if child.group == group_key -%} {%- set_global sub_routes = sub_routes | concat(with=[child_key]) -%} {%- endif -%} {%- endfor -%} {%- set sec_title = node.label[lang] -%} {%- set sec_pages = 0 -%} {%- set sec_subs = [] -%} {%- if tree.byRoute[route_key] is defined -%} {%- set sec = tree.byRoute[route_key] -%} {%- if sec.title is defined -%}{%- set sec_title = sec.title -%}{%- endif -%} {%- if sec.pages is defined -%}{%- set sec_pages = sec.pages -%}{%- endif -%} {%- if sec.subsections is defined -%}{%- set sec_subs = sec.subsections -%}{%- endif -%} {%- endif -%} {%- set has_children = sub_routes | length > 0 or sec_subs | length > 0 or sec_pages > 0 -%} {%- if has_children -%} {%- set sub_open = not node.collapsed or (mapped_url is starting_with(route_key)) -%}
{{ node.label[lang] }}
{%- if lang == 'zh' %}目录{% else %}Index{% endif -%} {%- if sec_pages > 0 -%} {%- if lang == 'zh' %}全部 {{ sec_pages }} 个页面{% else %}All {{ sec_pages }} pages{% endif -%} {%- endif -%} {%- for child_key in sub_routes -%} {%- set child = nav.routes[child_key] -%} {{ child.label[lang] }} {%- endfor -%} {%- for sub_key in sec_subs -%} {%- set sub_title = sub_key -%} {%- set sub_pages = 0 -%} {%- set sub_subs = [] -%} {%- if tree.byRoute[sub_key] is defined -%} {%- set sub_sec = tree.byRoute[sub_key] -%} {%- if sub_sec.title is defined -%}{%- set sub_title = sub_sec.title -%}{%- endif -%} {%- if sub_sec.pages is defined -%}{%- set sub_pages = sub_sec.pages -%}{%- endif -%} {%- if sub_sec.subsections is defined -%}{%- set sub_subs = sub_sec.subsections -%}{%- endif -%} {%- endif -%}
{{ sub_title }}
{%- if lang == 'zh' %}目录{% else %}Index{% endif -%} {%- if sub_pages > 0 -%} {%- if lang == 'zh' %}全部 {{ sub_pages }} 个页面{% else %}All {{ sub_pages }} pages{% endif -%} {%- endif -%} {%- for sub2_key in sub_subs -%} {%- set sub2_title = sub2_key -%} {%- set sub2_pages = 0 -%} {%- set sub2_subs = [] -%} {%- if tree.byRoute[sub2_key] is defined -%} {%- set sub2_sec = tree.byRoute[sub2_key] -%} {%- if sub2_sec.title is defined -%}{%- set sub2_title = sub2_sec.title -%}{%- endif -%} {%- if sub2_sec.pages is defined -%}{%- set sub2_pages = sub2_sec.pages -%}{%- endif -%} {%- if sub2_sec.subsections is defined -%}{%- set sub2_subs = sub2_sec.subsections -%}{%- endif -%} {%- endif -%}
{{ sub2_title }}
{%- if lang == 'zh' %}目录{% else %}Index{% endif -%} {%- if sub2_pages > 0 -%} {%- if lang == 'zh' %}全部 {{ sub2_pages }} 个页面{% else %}All {{ sub2_pages }} pages{% endif -%} {%- endif -%} {%- for sub3_key in sub2_subs -%} {%- set sub3_title = sub3_key -%} {%- if tree.byRoute[sub3_key] is defined and tree.byRoute[sub3_key].title is defined -%} {%- set sub3_title = tree.byRoute[sub3_key].title -%} {%- endif -%} {{ sub3_title }} {%- endfor -%}
{%- endfor -%}
{%- endfor -%}
{%- else -%} {{ node.label[lang] }} {%- endif -%} {%- endfor -%}
{%- endif -%} {%- endfor -%} {%- else -%}
{%- if lang == 'zh' %}站点导航{% else %}Site Navigation{% endif -%}
{%- if nav.routes["/"] is defined -%} {{ nav.routes["/"].label[lang] }} {%- endif -%} {%- for version in nav.meta.versions -%} {%- for lang_code in nav.meta.languages -%} {%- set route_key = "/" ~ version ~ "/" ~ lang_code ~ "/" -%} {%- if nav.routes[route_key] is defined -%} {{ nav.routes[route_key].label[lang] }} {%- endif -%} {%- endfor -%} {%- endfor -%} {%- if nav.routes["/versions/"] is defined -%} {{ nav.routes["/versions/"].label[lang] }} {%- endif -%}
{%- endif -%} {%- endmacro render_nav -%}
